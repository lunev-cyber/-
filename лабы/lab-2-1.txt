#include <iostream>

int main() {
    // Координаты векторов
    int a1, a2, b1, b2;

    // Ввод координат первого вектора
    std::cout << "Введите координаты первого вектора (x, y): ";
    std::cin >> a1 >> a2;

    // Ввод координат второго вектора
    std::cout << "Введите координаты второго вектора (x, y): ";
    std::cin >> b1 >> b2;

    // Вычисление векторного произведения в 2D
    int cross_product = a1 * b2 - a2 * b1;

    // Вывод результата
    std::cout << "Векторное произведение: " << cross_product << std::endl;

    return 0;
}
