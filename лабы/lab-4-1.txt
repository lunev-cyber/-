#include <iostream>
#include <iomanip> // Для установки точности вывода

int main() {
    int n;  // Вводимое число
    double sum = 0.0;  // Переменная для суммы

    // Ввод натурального числа n
    std::cout << "Введите натуральное число n: ";
    std::cin >> n;

    // Проверка на корректность ввода
    if (n <= 0) {
        std::cout << "Ошибка: n должно быть натуральным числом (n > 0)." << std::endl;
        return 1;
    }

    // Вычисление суммы ряда
    for (int i = 1; i <= n; ++i) {
        sum += 1.0 / (i * i);
    }

    // Вывод результата с точностью до 6 знаков после запятой
    std::cout << "Сумма ряда: " << std::fixed << std::setprecision(6) << sum << std::endl;

    return 0;
}
