#include <stdio.h>

int main() {
    int N;

    // Ввод размера массива
    printf("Введите размер массива: ");
    scanf("%d", &N);

    // Проверка на корректность размера массива
    if (N < 3) {
        printf("Размер массива должен быть >= 3 (чтобы были соседи).\n");
        return 1;
    }

    int arr[N];  // Объявление массива

    // Ввод элементов массива
    printf("Введите элементы массива:\n");
    for (int i = 0; i < N; i++) {
        scanf("%d", &arr[i]);
    }

    int last_local_max_index = -1;  // Инициализация индекса локального максимума

    // Поиск последнего локального максимума
    for (int i = 1; i < N - 1; i++) {  // Проверяем элементы от 1 до N-2
        if (arr[i] > arr[i - 1] && arr[i] > arr[i + 1]) {
            last_local_max_index = i;  // Сохраняем индекс текущего локального максимума
        }
    }

    // Вывод результата
    if (last_local_max_index != -1) {
        printf("Номер последнего локального максимума: %d\n", last_local_max_index);
    } else {
        printf("Локальные максимумы отсутствуют.\n");
    }

    return 0;
}
