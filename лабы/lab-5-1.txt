#include <stdio.h>

int main() {
    int n, i = 2;  // n - число для разложения, i - текущий делитель

    // Ввод числа
    printf("Введите натуральное число для разложения: ");
    scanf("%d", &n);

    // Проверка на корректность ввода
    if (n <= 1) {
        printf("Число должно быть больше 1.\n");
        return 1;
    }

    printf("Простые множители числа %d: ", n);

    // Разложение на простые множители
    while (n > 1) {
        if (n % i == 0) {      // Если n делится на i без остатка
            printf("%d ", i);  // Выводим делитель
            n /= i;           // Делим n на i
        } else {
            i++;  // Переходим к следующему делителю
        }
    }

    printf("\n");
    return 0;
}
