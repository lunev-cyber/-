# Инициализация переменных
n = 2 + 1  # (2 + 1) = 3
factorial = 1
i = 1  # Начинаем с 1

# Цикл с постусловием (имитация)
while True:
    factorial *= i  # Умножаем на текущий счетчик
    i += 1  # Увеличиваем счетчик
    
    if i > n:  # Условие выхода из цикла
        break

# Вывод результата
print(f"Факториал числа {n} равен {factorial}")
