# Ввод размера массива и элементов
N = int(input("Введите размер массива: "))

# Проверка на минимальный размер
if N < 3:
    print("Размер массива должен быть >= 3 (чтобы были соседи).")
else:
    # Ввод элементов массива
    arr = list(map(int, input("Введите элементы массива через пробел: ").split()))
    
    # Проверка корректности ввода
    if len(arr) != N:
        print("Количество элементов массива не совпадает с размером!")
    else:
        max_local_minimum = float('-inf')  # Начальное значение для максимального локального минимума

        # Поиск локальных минимумов
        for i in range(1, N - 1):  # От второго до предпоследнего элемента
            if arr[i] < arr[i - 1] and arr[i] < arr[i + 1]:
                max_local_minimum = max(max_local_minimum, arr[i])

        # Проверка результата
        if max_local_minimum == float('-inf'):
            print("Локальные минимумы отсутствуют.")
        else:
            print("Максимальный из локальных минимумов:", max_local_minimum)
