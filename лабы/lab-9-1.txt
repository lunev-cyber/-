#include <stdio.h>
#include <math.h> // Для использования функции sqrt

int main() {
    double a, h;  // Основание и высота треугольника
    double b, perimeter; // Боковая сторона и периметр

    // Ввод значений основания и высоты
    printf("Введите длину основания треугольника: ");
    scanf("%lf", &a);

    printf("Введите высоту, проведенную к основанию: ");
    scanf("%lf", &h);

    // Проверка корректности ввода
    if (a <= 0 || h <= 0) {
        printf("Ошибка: длина основания и высота должны быть положительными числами.\n");
        return 1;
    }

    // Вычисление боковой стороны по теореме Пифагора
    b = sqrt((a / 2) * (a / 2) + h * h);

    // Вычисление периметра
    perimeter = a + 2 * b;

    // Вывод результата
    printf("Периметр равнобедренного треугольника: %.2lf\n", perimeter);

    return 0;
}
