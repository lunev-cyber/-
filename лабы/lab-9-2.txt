import math

def rectangular_to_polar(x, y):
    """
    Функция для перевода прямоугольных координат (x, y) в полярные координаты (r, φ).
    :param x: Координата x
    :param y: Координата y
    :return: Кортеж (r, φ) - модуль радиус-вектора и угол φ в радианах
    """
    r = math.sqrt(x ** 2 + y ** 2)  # Вычисление r
    phi = math.atan2(y, x)          # Вычисление φ (с использованием atan2 для правильного угла)

    return r, phi

# Ввод значений координат x и y
x = float(input("Введите координату x: "))
y = float(input("Введите координату y: "))

# Получаем полярные координаты
r, phi = rectangular_to_polar(x, y)

# Вывод результата
print(f"Полярные координаты:")
print(f"r = {r:.4f}")
print(f"φ = {phi:.4f} радиан")
