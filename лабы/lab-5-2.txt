# Ввод начальной суммы и количества лет
S = float(input("Введите начальную сумму вклада (S): "))
n = int(input("Введите количество лет: "))

# Проверка на корректность ввода
if S <= 0 or n <= 0:
    print("Ошибка: сумма вклада и количество лет должны быть положительными числами.")
else:
    # Переменная для подсчета лет
    year = 0

    # Цикл с предусловием
    while year < n:
        S *= 1.03  # Увеличиваем сумму на 3%
        year += 1  # Увеличиваем счетчик лет

    # Вывод результата с округлением до двух знаков после запятой
    print(f"Сумма вклада через {n} лет: {S:.2f} рублей")
