#include <iostream>

int main() {
    int S;  // Заданная сумма цифр
    int number = 100; // Начальное трехзначное число

    // Ввод суммы цифр
    std::cout << "Введите натуральное число (сумма цифр): ";
    std::cin >> S;

    // Проверка на корректность ввода
    if (S < 1 || S > 27) { // Максимальная сумма цифр для трехзначного числа: 9 + 9 + 9 = 27
        std::cout << "Ошибка: сумма цифр должна быть в диапазоне от 1 до 27." << std::endl;
        return 1;
    }

    // Цикл с постусловием для перебора трехзначных чисел
    do {
        int sum = (number / 100) + ((number / 10) % 10) + (number % 10); // Сумма цифр числа
        if (sum == S) {
            std::cout << number << " "; // Выводим число, если сумма совпадает
        }
        number++; // Переходим к следующему числу
    } while (number <= 999); // Проверка условия

    std::cout << std::endl;
    return 0;
}
