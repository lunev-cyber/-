def find_elements_greater_than_right(arr):
    """
    Находит номера элементов, которые больше своего правого соседа.
    :param arr: Входной массив
    :return: Список номеров элементов и их количество
    """
    indices = []  # Список для хранения номеров подходящих элементов
    for i in range(len(arr) - 1):  # Проходим по элементам, кроме последнего
        if arr[i] > arr[i + 1]:    # Проверка условия
            indices.append(i)      # Добавляем индекс элемента
    return indices, len(indices)   # Возвращаем список номеров и количество

def input_array():
    """
    Ввод массива от пользователя.
    :return: Список чисел
    """
    n = int(input("Введите размер массива: "))
    print("Введите элементы массива через пробел:")
    return list(map(int, input().split()))

# Основная программа
def main():
    # Ввод массива
    arr = input_array()

    # Поиск номеров элементов и их количества
    indices, count = find_elements_greater_than_right(arr)

    # Вывод результатов
    print("\nНомера элементов, которые больше своего правого соседа:")
    print(*indices)  # Выводим индексы через пробел
    print(f"Количество таких элементов: {count}")

# Запуск программы
if __name__ == "__main__":
    main()
